#pragma warning(disable:4996)
#include <iostream>
#include<string>

int main()
{
   std::string input = "(((()(()()))(())()))(()())";
   std::cin >> input;
   int sum = 0;
   int stick =0;
   for (int i = 0; i < input.size() - 1; i++)
   {
      if (input[i] == '(' && input[i + 1] == ')')
      {
         sum += stick;
         i++;
      }
      else if (input[i] == '(')
      {
         stick++;
      }
      else
      {
         stick--;
         sum++;
      }
   }
   if (stick) sum++;
   std::cout << sum << std::endl;
   return 0;
}
